CREATE TABLE [Component] (
[Id] INTEGER  PRIMARY KEY AUTOINCREMENT NOT NULL,
[CLRType] NVARCHAR(256)  UNIQUE NOT NULL,
[Name] NVARCHAR(64)  NULL,
[Description] TEXT  NULL,
[Dependencies] TEXT  NULL
);$

CREATE TABLE [ViewType] (
[Id] INTEGER  PRIMARY KEY AUTOINCREMENT NOT NULL,
[Name] VARCHAR(64)  NOT NULL
);$

CREATE TABLE [Window] (
  [Id] INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT, 
  [Name] VARCHAR(64) NOT NULL, 
  [AppTypeId] Int NOT NULL);$

CREATE TABLE [View] (
  [Id] INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT, 
  [Name] VARCHAR(64) NOT NULL UNIQUE, 
  [IconPath] VARCHAR(256), 
  [ViewTypeId] INTEGER NOT NULL, 
  [CLRType] VARCHAR(256) NOT NULL DEFAULT 'Altus.Core.Presentation.ViewModels.View,Core');$

CREATE TABLE [WPFTemplate] (
[Id] INTEGER  PRIMARY KEY AUTOINCREMENT NOT NULL,
[Name] VARCHAR(64)  NOT NULL,
[XAML] TEXT  NOT NULL,
[Size] VARCHAR(64)  NOT NULL,
[MasterTemplateId] INTEGER  NULL,
[IsMaster] BOOLEAN DEFAULT '0' NOT NULL
);$

CREATE TABLE [WPFTemplateView] (
[WPFTemplateId] INTEGER  NULL,
[ViewId] INTEGER  NOT NULL,
PRIMARY KEY ([WPFTemplateId],[ViewId])
);$

CREATE TABLE [WindowView] (
[WindowId] INTEGER  NOT NULL,
[ViewId] INTEGER  NOT NULL,
PRIMARY KEY ([WindowId],[ViewId]));$

CREATE TABLE [AppType] (
[Id] INTEGER  PRIMARY KEY AUTOINCREMENT NOT NULL,
[Name] VARCHAR(32)  NOT NULL
);$
